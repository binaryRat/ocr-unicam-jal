run main.py <input_dir> <output_dir> <options>

options:
 -t --tresholding : applies the adaptive threshold on the input images and save them in the output dir
 -e --edgedetection : applies the edge detection on the input images and save them in the output dir
 -o --ocr : applies the ocr algorithm on the input images, if one of the denoising technique is selected, the script will work on the denoised images
